# Python dependencies for upload_to_yakyak.py
# Install:  python3 -m pip install -r requirements.txt
#
# yakyak-sdk is the OpenAPI-generated YakYak client (https://pypi.org/project/yakyak-sdk/).
# It is the only third-party import; everything else the script uses is stdlib.
# Its runtime deps (urllib3, python-dateutil, pydantic, typing-extensions) are
# pulled in automatically by pip.
# 0.0.7+ models the request bodies the engine needs (switch-campaign-mode,
# export-render, set-soundtrack-audio, set-movie-metadata.plot, …), so those
# calls use the typed SDK methods instead of raw transport.
yakyak-sdk==0.0.7

# certifi gives us a guaranteed CA bundle so TLS verification works on a fresh
# python.org macOS install without running "Install Certificates.command".
# The script pins all HTTPS calls to certifi.where() (see CA_BUNDLE in the script).
certifi>=2024.0.0
